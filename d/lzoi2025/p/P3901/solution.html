    • 时间复杂度：O(nn)O(n\sqrt{n})O(nn​)
    • 期望得分：100 分
    #include <bits/stdc++.h>
using namespace std;

struct Node {
    int l, r, id;    
};

int main() {
    ios::sync_with_stdio(false);
    cin.tie(0); cout.tie(0);
    
    int n, m;
    cin >> n >> m;
    
    vector<int> a(n + 1);
    for (int i = 1; i <= n; i++) cin >> a[i];
    
    vector<Node> q(m);
    for (int i = 0; i < m; i++) cin >> q[i].l >> q[i].r, q[i].id = i;
    
    int d = sqrt(n);
    sort(q.begin(), q.end(), [&](const Node& x, const Node& y) {
        // 将询问的左端点, 每块长为 d 编号, 编号小的排前面
        if (x.l / d != y.l / d) return x.l / d < y.l / d;
        
        // 右端点 r 奇偶排序优化, 预计快 30%
        // 左端点块号为奇数, 右端点从小到大排序
        if (x.l / d & 1) return x.r < y.r;
        
        // 左端点块号偶奇数, 右端点从大到小排序
        return x.r > y.r;
    });
    
    int tol = 0;                // 不同数个数
    int L = 1, R = 0;           // 当前处理的区间, 初始是空的
    vector<int> cnt(n + 1, 0);  // cnt[i]: i 出现的次数
    vector<bool> ans(m);
    
    for (int i = 0; i < m; i++) {
        int l = q[i].l, r = q[i].r, id = q[i].id;
        
        // 右边界 R 往右扩展
        while (R < r) tol += (++cnt[a[++R]] == 1);
        
        // 左边界 L 往左扩展
        while (L > l) tol += (++cnt[a[--L]] == 1);
        
        // 右边界 R 往左缩小
        while (R > r) tol -= (--cnt[a[R--]] == 0);
        
        // 左边界 L 往右缩小
        while (L < l) tol -= (--cnt[a[L++]] == 0);
        
        // 处理完后记录答案
        ans[id] = (tol == r - l + 1);
    }
    for (int i = 0; i < m; i++) cout << (ans[i] ? "Yes" : "No") << '\n';
    
    return 0;
}

            #include <bits/stdc++.h>
using namespace std;

int main() {

    cin.tie(0); cout.tie(0);
    
    int n, m, l, r;

        cin >> l >> r;
        
        int tol = 0;                // 不同数个数
