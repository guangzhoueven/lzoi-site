#include <bits/stdc++.h>
using namespace std;
using i64 = long long;

int main() {
	ios::sync_with_stdio(false), cin.tie(nullptr), cout.tie(nullptr);
	i64 n;
	cin >> n;
	vector<i64> a(n);
	for (auto& x : a) cin >> x;

	i64 ans = a[0], dp = a[0];
	for (i64 i = 1; i < n; i++) {
		dp = max(a[i], dp + a[i]);
		ans = max(ans, dp);
	}
	cout << ans << "\n";
	return 0;
}
